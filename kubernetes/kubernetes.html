Kubernetes Command Refference

üìù Cluster Setup & Info

# Create a new cluster (using Minikube as an example)
minikube start

# Show cluster info
kubectl cluster-info

# Show nodes in the cluster
kubectl get nodes

# View the status of the cluster
kubectl get cs

üì¶ Pods

# List all pods in the default namespace
kubectl get pods

# List all pods in all namespaces
kubectl get pods --all-namespaces

# Describe a specific pod
kubectl describe pod pod-name

# Delete a pod
kubectl delete pod pod-name

üìë Deployments

# List deployments
kubectl get deployments

# Create a deployment
kubectl create deployment deployment-name --image=image-name

# Scale a deployment
kubectl scale deployment deployment-name --replicas=3

# Update a deployment with a new image
kubectl set image deployment/deployment-name container-name=image-name

# Rollback to previous deployment
kubectl rollout undo deployment/deployment-name

üìÅ Services

# List services
kubectl get svc

# Expose deployment as a service
kubectl expose deployment deployment-name --type=LoadBalancer --port=8080

# Get detailed service information
kubectl describe svc service-name

üí° Namespaces

# List all namespaces
kubectl get namespaces

# Create a new namespace
kubectl create namespace namespace-name

# Set the current namespace for kubectl
kubectl config set-context --current --namespace=namespace-name

üîß ConfigMaps & Secrets

# Create a configmap
kubectl create configmap configmap-name --from-literal=key=value

# Create a secret
kubectl create secret generic secret-name --from-literal=key=value

# List configmaps
kubectl get configmaps

# List secrets
kubectl get secrets

üîÑ Logs & Troubleshooting

# Get logs for a pod
kubectl logs pod-name

# Get logs for a specific container in a pod
kubectl logs pod-name -c container-name

# Follow logs in real-time
kubectl logs -f pod-name

# Show cluster events
kubectl get events

üîë RBAC & Access Control

# View roles
kubectl get roles

# Create a role
kubectl create role role-name --verb=get,list --resource=pods --namespace=namespace-name

# Bind role to user
kubectl create rolebinding binding-name --role=role-name --user=user-name --namespace=namespace-name
