🔧 Installation & Setup

sudo apt install docker.io
sudo systemctl start docker
sudo systemctl enable docker
sudo usermod -aG docker $USER

📦 Images

docker pull nginx
docker build -t myimage .
docker images
docker rmi image_id
docker rmi -f image_id
docker save -o myimage.tar myimage
docker load -i myimage.tar

📂 Containers

docker run hello-world
docker run -it ubuntu bash
docker run -d --name mynginx -p 8080:80 nginx
docker run --rm busybox echo "Hello!"
docker ps
docker ps -a
docker stop container_id
docker start container_id
docker restart container_id
docker rm container_id
docker rm -f container_id
docker logs container_id
docker exec -it container_id bash
docker attach container_id
docker inspect container_id
docker run --name  -p host-port:container-port -v /home/docker/host:/container/app --network 
docker run --name mycontainer -p 8080:80 -v /host:/container --network mynet --env VAR=value --restart unless-stopped --memory="512m" --cpus="1.5" myimage

🗃 Volumes

docker volume create myvol
docker volume ls
docker volume inspect myvol
docker volume rm myvol
docker run -v myvol:/data ubuntu

🌐 Networks

docker network ls
docker network create mynet
docker network inspect mynet
docker network connect mynet container
docker run --network mynet ...

🧪 Dockerfile Example

FROM ubuntu
RUN apt update && apt install -y nginx
CMD ["nginx", "-g", "daemon off;"]

📜 Docker Compose

docker-compose up
docker-compose up -d
docker-compose down
docker-compose build
docker-compose logs
docker-compose exec service_name bash

🚀 Full docker run Example

docker run \
  --name mycontainer \
  -p 8080:80 \
  -v /host:/container \
  --network mynet \
  --env VAR=value \
  --restart unless-stopped \
  --memory="512m" \
  --cpus="1.5" \
  myimage

📤 Export & Import

docker export container_id > container.tar
docker import container.tar mycontainer

🧹 Clean Up

docker system prune
docker image prune
docker container prune
docker volume prune
